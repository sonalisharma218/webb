JavaScript Function Programs
1. Filter Out Negative Numbers
function filterNegatives(arr) {
return arr.filter(num => num >= 0);
}
console.log(filterNegatives([1, -2, 3, -4, 5]));
2. Difference from 13 or Double if Greater
function diffFrom13(n) {
return n > 13 ? 2 * Math.abs(n - 13) : 13 - n;
}
console.log(diffFrom13(20));
3. Fibonacci Series Function
function fibonacci(n) {
let a = 0, b = 1, result = [a, b];
for (let i = 2; i < n; i++) {
let next = a + b;
result.push(next);
a = b;
b = next;
}
return result;
}
console.log(fibonacci(10));
4. Sum of Multiples of 3 and 5 Under 1000
function sumMultiples() {
let sum = 0;
for (let i = 0; i < 1000; i++) {
if (i % 3 === 0 || i % 5 === 0) sum += i;
}
return sum;
}
console.log(sumMultiples());
